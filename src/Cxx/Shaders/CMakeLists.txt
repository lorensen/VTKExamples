project (${WIKI}Shaders)

if(NOT WikiExamples_BINARY_DIR)
find_package(VTK REQUIRED)
  include(${VTK_USE_FILE})
endif()

set(KIT_LIBS ${VTK_LIBRARIES})
#
# Build all .cxx files in the directory
file(GLOB ALL_FILES *.cxx)

# VTK follows the C++03 standard but some examples require C++11
include(${WikiExamples_SOURCE_DIR}/CMake/RequiresCxxVersion.cmake)
Requires_CxxVersion(SpatterShader 11 ALL_FILES)

include(${WikiExamples_SOURCE_DIR}/CMake/RequiresVersion.cmake)
Requires_Version(BozoShader "8.0.0" ALL_FILES)
Requires_Version(BozoShaderDemo "8.0.0" ALL_FILES)
Requires_Version(MarbleShader "8.0.0" ALL_FILES)
Requires_Version(MarbleShaderDemo "8.0.0" ALL_FILES)
Requires_Version(ColorByNormal "8.0.0" ALL_FILES)
Requires_Version(CubeMap "8.1.0" ALL_FILES)
Requires_Version(SphereMap "8.2.0" ALL_FILES)

foreach(SOURCE_FILE ${ALL_FILES})
  string(REPLACE ".cxx" "" TMP ${SOURCE_FILE})
  string(REPLACE ${CMAKE_CURRENT_SOURCE_DIR}/ "" EXAMPLE ${TMP})
  add_executable(${WIKI}${EXAMPLE} ${EXECUTABLE_FLAG} ${EXAMPLE}.cxx)
  target_link_libraries(${WIKI}${EXAMPLE} ${KIT_LIBS})
endforeach()
